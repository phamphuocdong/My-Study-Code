#!/bin/bash

#Update software on the instance
sudo yum update -y

#Install Apache web server
sudo yum install -y httpd

#Start the web server
sudo systemctl start httpd

#Enable the web server to start when ever system boost
sudo systemctl enable httpd

# Go to html directory
cd /var/www/html

#Install the static website source code
sudo wget https://publicsourcecodefordownloading.s3.amazonaws.com/StaticWeb.zip

#Unzip the code
sudo unzip StaticWeb.zip
